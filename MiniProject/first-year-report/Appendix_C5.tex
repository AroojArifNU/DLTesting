%----- Mutatation operator ------%
function  anew=mutate(a,insite)
    nn=length(a);
    for i=1:insite
    j=floor(rand*nn);
        if j==0
            j=randi(nn,1,1);
        end
    k=a(j);
    if k==1
        a(j)=0;
    else
       if k==0
        a(j)=1;
       end
    end
    anew=a;
   end
